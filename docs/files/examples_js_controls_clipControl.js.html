/**

 Helper that visualizes the position and direction of a plane.

 @class ClipControl
 @constructor
 @param cfg {*} Configuration
 @param cfg.cameraControl {BIMCameraControl} The camera input control.
 @param [cfg.pos=[0,0,0]] {Float32Array} World-space position.
 @param [cfg.dir=[0,0,1]] {Float32Array} World-space direction vector.
 @param [cfg.color=[0.4,0.4,0.4]] {Float32Array} Emmissive color
 @param [cfg.solid=true] {Boolean} Indicates whether or not this helper is filled with color or just wireframe.
 @param [cfg.visible=true] {Boolean} Indicates whether or not this helper is visible.
 @param [cfg.planeSize] {Float32Array} The width and height of the ClipControl plane indicator.
 @param [cfg.autoPlaneSize=false] {Boolean} Indicates whether or not this ClipControl's
 {{#crossLink "ClipControl/planeSize:property"}}{{/crossLink}} is automatically sized to fit within
 the {{#crossLink "Scene/aabb:property"}}Scene's boundary{{/crossLink}}.
 */
(function () {

    "use strict";

    xeogl.ClipControl = xeogl.Component.extend({

        type: "xeogl.ClipControl",

        _init: function (cfg) {

            this._cameraControl = cfg.cameraControl;

            this._solid = false; // Whether to render clip plane with a fill color.
            this._visible = false; // Whether or not the GumBall is visible.
            this._clipStartDir = xeogl.math.vec3(); // Saves direction of clip plane when we start dragging a GumBall arrow or hoop.
            this._clipPos = xeogl.math.vec3(); // Holds the current position of the center of the clip plane.

            this._planeScale = null; // xeogl.Scale to size the clip plane helper
            this._gumballQuaternion = null; // xeogl.Quaternion to rotate the GumBall
            this._gumballTranslate = null; // xeogl.Translate to position the GumBall

            this._displayEntities = null; // Map of xeogl.Entity's that are always visible when GumBall visible
            this._affordanceEntities = null; // Map of xeogl.Entity's that are shown momentarily to give "affordance" when interacting with the GumBall

            this._initEntities();

            // Configure

            this.planeSize = cfg.planeSize;
            this.autoPlaneSize = cfg.autoPlaneSize;
            this.color = cfg.color;
            this.solid = cfg.solid;
            this.clip = cfg.clip;
            this.visible = cfg.visible;
            this.active = true;

            this._initEvents();
        },

        _update: (function () {
            var arrowPositions = new Float32Array(6);
            return function () {

                var pos = this._pos;
                var dir = this._dir;

                // Rebuild arrow geometry

                arrowPositions[0] = pos[0];
                arrowPositions[1] = pos[1];
                arrowPositions[2] = pos[2];
                arrowPositions[3] = pos[0] + dir[0];
                arrowPositions[4] = pos[1] + dir[1];
                arrowPositions[5] = pos[2] + dir[2];

                //this._displayEntities.arrow.geometry.positions = arrowPositions;
            }
        })(),

        _props: {

            /**
             Indicates whether this ClipControl is active or not.

             When inactive, the ClipControl will ignore input events.

             @property active
             @default true
             @type Boolean
             */
            active: {

                set: function (value) {
                    value = value !== false;
                    if (value === this._active) {
                        return;
                    }
                    this._active = value;
                },

                get: function () {
                    return this._active;
                }
            },

            /**
             The {{#crossLink "Clip"}}{{/crossLink}} controlled by this ClipControl.

             @property clip
             @type Clip
             */
            clip: {

                set: (function () {

                    var zeroVec = new Float32Array([0, 0, -1]);
                    var quat = new Float32Array(4);

                    return function (value) {

                        this._attach({name: "clip", type: "xeogl.Clip", component: value});

                        var clip = this._attached.clip;

                        if (clip) {

                            // Save initial clip plane pos and direction,
                            // Set GumBall's current direction and position

                            this._clipStartDir.set(clip.dir);
                            xeogl.math.vec3PairToQuaternion(zeroVec, this._clipStartDir, quat);
                            this._gumballQuaternion.xyzw = quat;

                            this._clipPos.set(clip.pos);
                            this._gumballTranslate.xyz = this._clipPos;
                        }
                    };
                })(),

                get: function () {
                    return this._attached.clip;
                }
            },

            /**
             * The width and height of this ClipControl's plane indicator.
             *
             * Values assigned to this property will be overridden by an auto-computed value when
             * {{#crossLink "ClipControl/autoPlaneSize:property"}}{{/crossLink}} is true.
             *
             * @property planeSize
             * @default [1,1]
             * @type {Float32Array}
             */
            planeSize: {

                set: function (value) {
                    (this._planeSize = this._planeSize || new xeogl.math.vec2()).set(value || [1, 1]);
                    this._planeScale.xyz = [this._planeSize[0], this._planeSize[1], 1.0];
                },

                get: function () {
                    return this._planeSize;
                }
            },

            /**
             * Indicates whether this ClipControl's {{#crossLink "ClipControl/planeSize:property"}}{{/crossLink}} is automatically
             * generated or not.
             *
             * When auto-generated, {{#crossLink "ClipControl/planeSize:property"}}{{/crossLink}} will automatically size
             * to fit within the {{#crossLink "Scene/aabb:property"}}Scene's boundary{{/crossLink}}.
             *
             * @property autoPlaneSize
             * @default false
             * @type {Boolean}
             */
            autoPlaneSize: {

                set: function (value) {

                    value = !!value;

                    if (this._autoPlaneSize === value) {
                        return;
                    }

                    this._autoPlaneSize = value;

                    if (this._autoPlaneSize) {
                        if (!this._onSceneAABB) {
                            this._onSceneAABB = this.scene.on("boundary", function () {
                                var aabbDiag = xeogl.math.getAABB3Diag(this.scene.aabb);
                                var clipSize = (aabbDiag * 0.50);
                                this.planeSize = [clipSize, clipSize];
                            }, this);
                        }
                    } else {
                        if (this._onSceneAABB) {
                            this.scene.off(this._onSceneAABB);
                            this._onSceneAABB = null;
                        }
                    }
                },

                get: function () {
                    return this._autoPlaneSize;
                }
            },

            /**
             * The color of this ClipControl.
             *
             * @property color
             * @default [0.4,0.4,0.4]
             * @type {Float32Array}
             */
            color: {

                set: function (value) {
                    (this._color = this._color || new xeogl.math.vec3()).set(value || [0.4, 0.4, 0.4]);
                    this._displayEntities.planeWire.material.emissive = this._color;
                },

                get: function () {
                    return this._color;
                }
            },

            /**
             Indicates whether this ClipControl is filled with color or just wireframe.

             @property solid
             @default true
             @type Boolean
             */
            solid: {

                set: function (value) {
                    this._solid = value !== false;
                    this._displayEntities.planeSolid.visible = this._solid && this._visible;
                },

                get: function () {
                    return this._solid;
                }
            },

            /**
             * Visibility of this ClipControl.
             *
             * @property visible
             * @type Boolean
             * @default true
             */
            visible: {

                set: function (value) {
                    value = !!value;
                    if (this._visible === value) {
                        return;
                    }
                    this._visible = value;
                    var id;
                    for (id in this._displayEntities) {
                        if (this._displayEntities.hasOwnProperty(id)) {
                            this._displayEntities[id].visible = value;
                        }
                    }
                    if (!value) { // Affordance entities are not initially visible
                        for (id in this._affordanceEntities) {
                            if (this._affordanceEntities.hasOwnProperty(id)) {
                                this._affordanceEntities[id].visible = value;
                            }
                        }
                    }
                },

                get: function () {
                    return this._visible;
                }
            }
        },

            var dragAction = null; // Action we're doing while we drag an arrow or hoop.

            var dragClipPosOrigin = math.vec3();
            var dragCanvasOrigin = math.vec2();
            var dragCanvasVec = math.vec2();

            var xLocalAxis = math.vec3([1, 0, 0]);
            var yLocalAxis = math.vec3([0, -1, 0]);
            var zLocalAxis = math.vec3([0, 0, 1]);

            var rotateWorldVec = math.vec3();
            var panWorldVec = math.vec3();
            var panCanvasVec = math.vec2();

            canvas.oncontextmenu = function (e) {
                e.preventDefault();
            };

            var getClickCoordsWithinElement = (function () {
                var coords = new Float32Array(2);
                return function (event) {
                    if (!event) {
                        event = window.event;
                        coords[0] = event.x;
                        coords[a] = event.y;
                    } else {
                        var element = event.target;
                        var totalOffsetLeft = 0;
                        var totalOffsetTop = 0;

                        while (element.offsetParent) {
                            totalOffsetLeft += element.offsetLeft;
                            totalOffsetTop += element.offsetTop;
                            element = element.offsetParent;
                        }
                        coords[0] = event.pageX - totalOffsetLeft;
                        coords[1] = event.pageY - totalOffsetTop;
                    }
                    return coords;
                };
            })();

            var localToWorldVec = (function () {
                var math = xeogl.math;
                var mat = math.mat4();
                return function (localVec, worldVec) {
                    math.quaternionToMat4(self._gumballQuaternion.xyzw, mat);
                    math.transformVec3(mat, localVec, worldVec);
                    return worldVec;
                };
            })();

            var worldToCanvasVec = (function () {
                var math = xeogl.math;
                var tempVec4a = math.vec4([0, 0, 0, 1]);
                var tempVec4b = math.vec4();
                var tempVec4c = math.vec4();
                return function (worldVec, canvasVec) {
                    tempVec4a.set(worldVec);
                    math.transformPoint4(self.scene.camera.matrix, tempVec4a, tempVec4b);
                    math.transformPoint4(self.scene.camera.projMatrix, tempVec4b, tempVec4c);
                    var aabb = self.scene.canvas.boundary;
                    canvasVec[0] = Math.floor((1 + tempVec4c[0] / tempVec4c[3]) * aabb[2] / 2);
                    canvasVec[1] = Math.floor((1 - tempVec4c[1] / tempVec4c[3]) * aabb[3] / 2);
                    canvasVec[0] -= aabb[2] / 2;
                    canvasVec[1] -= aabb[3] / 2;
                    return canvasVec;
                };
            })();

            var pick = (function () {

                var lastHighlightedEntity;
                var lastShownEntity;

                return function pick(canvasPos) {

                    var hit = scene.pick({
                        canvasPos: canvasPos,
                        entities: [
                            this._displayEntities.xArrow,
                            this._displayEntities.yArrow,
                            this._displayEntities.zArrow,
                            this._displayEntities.xCurvePickable,
                            this._displayEntities.yCurvePickable

                    });

                    if (lastHighlightedEntity) {
                        lastHighlightedEntity.highlight = false;
                    }

                    if (lastShownEntity) {
                        lastShownEntity.visible = false;

                    if (hit) {

                        var id = hit.entity.id;

                        var highlightEntity;
                        var shownEntity;

                        switch (id) {
                            case self._displayEntities.xArrow.id:
                                console.log("picked X arrow");

                                nextDragAction = DRAG_ACTIONS.xPan;
                                localToWorldVec(xLocalAxis, panWorldVec);
                                worldToCanvasVec(panWorldVec, panCanvasVec);
                                break;

                            case self._displayEntities.yArrow.id:
                                console.log("picked Y arrow");
                                //highlightEntity = self._displayEntities.yArrow;
                                nextDragAction = DRAG_ACTIONS.yPan;
                                localToWorldVec(yLocalAxis, panWorldVec);
                                worldToCanvasVec(panWorldVec, panCanvasVec);
                                break;

                            case self._displayEntities.zArrow.id:
                                console.log("picked Z arrow");
                                //highlightEntity = self._displayEntities.zArrow;
                                nextDragAction = DRAG_ACTIONS.zPan;
                                localToWorldVec(zLocalAxis, panWorldVec);
                                worldToCanvasVec(panWorldVec, panCanvasVec);
                                break;

                            case self._displayEntities.xCurvePickable.id:
                                //highlightEntity = self._displayEntities.xCurve;
                                localToWorldVec(xLocalAxis, rotateWorldVec);
                                shownEntity = self._affordanceEntities.xHoop;
                                nextDragAction = DRAG_ACTIONS.yRotate;
                                break;

                            case self._displayEntities.yCurvePickable.id:
                                //highlightEntity = self._displayEntities.yCurve;
                                localToWorldVec(yLocalAxis, rotateWorldVec);
                                shownEntity = self._affordanceEntities.yHoop;
                                nextDragAction = DRAG_ACTIONS.xRotate;
                                break;

                            // case self._displayEntities.zCurvePickable.id:

                        if (highlightEntity) {
                            highlightEntity.highlight = true;
                        }

                        if (shownEntity) {
                            shownEntity.visible = true;
                        }

                        lastHighlightedEntity = highlightEntity;
                        lastShownEntity = shownEntity;

                    } else {

                        lastHighlightedEntity = null;
                        lastShownEntity = null;
                        nextDragAction = DRAG_ACTIONS.none;
                    }
                };

                    yDelta = 0;

                canvas.addEventListener("mouseleave", function () {

                    if (!self._active) {
                        return;
                    }

                    over = false;

                    xDelta = 0;
                    yDelta = 0;
                });

                canvas.addEventListener("mousedown", function (e) {

                    if (!self._active) {
                        return;
                    }

                    if (!over) {
                        return;
                    }

                    switch (e.which) {

                        case 1: // Left button

                            mouseDownLeft = true;
                            down = true;
                            xDelta = 0;
                            yDelta = 0;
                            var coords = getClickCoordsWithinElement(e);
                            lastX = coords[0];
                            lastY = coords[1];

                            dragAction = nextDragAction;

                            dragCanvasOrigin[0] = coords[0];
                            dragCanvasOrigin[1] = coords[1];

                            dragClipPosOrigin.set(self._clipPos);

                            break;

                        default:
                            break;
                    }
                });

                canvas.addEventListener("mousemove", function (e) {

                    if (!self._active) {
                        return;
                    }

                    if (!over) {
                        return;
                    }

                    var coords = getClickCoordsWithinElement(e);

                    if (!down) {
                        pick(coords);
                        return;
                    }

                    var x = coords[0];
                    var y = coords[1];

                    xDelta += (x - lastX) * 0.1;
                    yDelta += (y - lastY) * 0.1;

                    lastX = x;
                    lastY = y;

                    dragCanvasVec[0] = coords[0] - dragCanvasOrigin[0];
                    dragCanvasVec[1] = coords[1] - dragCanvasOrigin[1];
                });

                canvas.addEventListener("mouseup", function (e) {

                    if (!self._active) {
                        return;
                    }

                    switch (e.which) {
                        case 1: // Left button
                            mouseDownLeft = false;
                            break;
                        case 2: // Middle/both buttons
                            mouseDownMiddle = false;
                            break;
                        case 3: // Right button
                            mouseDownRight = false;
                            break;
                        default:
                            break;
                    }

                    down = false;

                    xDelta = 0;
                    yDelta = 0;
                });

                var pan = function (inc, f) {
                    self._clipPos[0] = dragClipPosOrigin[0] + inc[0] * f;
                    self._clipPos[1] = dragClipPosOrigin[1] + inc[1] * f;
                    self._clipPos[2] = dragClipPosOrigin[2] + inc[2] * f;
                    self._gumballTranslate.xyz = self._clipPos;
                    if (self._attached.clip) {
                        self._attached.clip.pos = self._clipPos;
                    }
                };

                var rotate = function (axis, incDegrees) {
                    self._gumballQuaternion.rotate([axis[0], axis[1], axis[2], incDegrees]);
                    rotateClip();
                };

                var rotateClip = (function () { // <<<<<<<------- suspect
                    var math = xeogl.math;
                    var dir = math.vec3();
                    var mat = math.mat4();
                    var zeroVec = new Float32Array([0, 0, -1]);
                    return function () {
                        if (self._attached.clip) {
                            math.quaternionToMat4(self._gumballQuaternion.xyzw, mat);  // << ---
                            math.transformVec3(mat, zeroVec, dir);
                            self._attached.clip.dir = dir;
                        }
                    };
                })();

                var lastDragLen = 0;

                scene.on("tick", function () {

                    // Periodically checking if there's anything we should be doing to move the GumBall

                    if (dragAction === DRAG_ACTIONS.none) {
                        return;
                    }

                    if (xDelta === 0 && yDelta === 0) {
                        return;
                    }

                    if (dragAction === DRAG_ACTIONS.xPan || dragAction === DRAG_ACTIONS.yPan || dragAction === DRAG_ACTIONS.zPan) {

                        // Compute panRate from dragCanvasVec and panCanvasVec

                        var panVec3 = math.vec3([panCanvasVec[0], panCanvasVec[1], 0.0]);

                        var dragVec3 = math.vec3([dragCanvasVec[0], dragCanvasVec[1], 0.0]);

                        math.normalizeVec3(panVec3);
                        math.normalizeVec3(dragVec3);

                        var angleRad = Math.atan2(panCanvasVec[1], panCanvasVec[0]) - Math.atan2(dragCanvasVec[1], dragCanvasVec[0]);
                        var angleDeg = angleRad / 0.0174533;
                        var radiaans = Math.PI * 2;

                        var dragLen = math.lenVec2(dragCanvasVec);
                        var panRate = dragLen * Math.sin(angleRad - (radiaans / 4));

                        panRate *= 0.015;

                        console.log("panRate = " + panRate);

                        // axisWire.setStartAndEnd(200, 200, 200 + panCanvasVec[0], 200 + panCanvasVec[1]);

                        console.log("dragLen = " + dragLen);

                        // panRate *= 0.1;

                        switch (dragAction) {
                            case DRAG_ACTIONS.xPan:
                                pan(panWorldVec, -panRate);
                                break;
                            case DRAG_ACTIONS.yPan:
                                pan(panWorldVec, -panRate);
                                break;
                            case DRAG_ACTIONS.zPan:
                                pan(panWorldVec, -panRate);
                                break;
                        }

                    } else {


                        var dragLen = math.lenVec2(dragCanvasVec);
                        var rotateRate = (dragLen - lastDragLen) * 0.1;

                        console.log("dragLen=" + dragLen)
                        console.log("rotateRate=" + rotateRate)

                        switch (dragAction) {
                            case DRAG_ACTIONS.xRotate:
                                rotate(xLocalAxis, rotateRate);
                                break;
                            case DRAG_ACTIONS.yRotate:
                                rotate(yLocalAxis, rotateRate);
                                break;
                        }

                        lastDragLen = dragLen;
                    }


                    xDelta = 0;
                    yDelta = 0;
                });

            })();
        },

        _destroy: function () {
