/**

 Helper that visualizes the position and direction of a plane.

 @class Ruler
 @constructor
 @param cfg {*} Configuration
 @param [cfg.pos=[0,0,0]] {Float32Array} World-space position.
 @param [cfg.dir=[0,0,1]] {Float32Array} World-space direction vector.
 @param [cfg.color=[0.4,0.4,0.4]] {Float32Array} Emmissive color
 @param [cfg.visible=true] {Boolean} Indicates whether or not this helper is visible.
 @param [cfg.planeSize] {Float32Array} The width and height of the Ruler plane indicator.
 @param [cfg.autoPlaneSize=false] {Boolean} Indicates whether or not this Ruler's
 {{#crossLink "Ruler/planeSize:property"}}{{/crossLink}} is automatically sized to fit within
 the {{#crossLink "Scene/worldBoundary:property"}}Scene's worldBoundary{{/crossLink}}.
 */
(function () {

    "use strict";

    function Wire(parentElement, cfg) {

        cfg = cfg || {};

        this._wire = document.createElement('div');
        var wire = this._wire;
        var wireStyle = wire.style;
        wireStyle.border = "solid 2px " + (cfg.color || "black");
        wireStyle.position = "absolute";
        wireStyle["z-index"] = "2000001";
        wireStyle.width = 0 + "px";
        wireStyle.height = 0 + "px";
        wireStyle.visibility = "visible";
        wireStyle.top = 0 + "px";
        wireStyle.left = 0 + "px";
        wireStyle["pointer-events"] = "none";
        wireStyle['-webkit-transform'] = 'rotate(0deg)';
        wireStyle['-moz-transform'] = 'rotate(0deg)';
        wireStyle['-ms-transform'] = 'rotate(0deg)';
        wireStyle['-o-transform'] = 'rotate(0deg)';
        wireStyle['transform'] = 'rotate(0deg)';
        parentElement.appendChild(wire);

        this._x1 = 0;
        this._y1 = 0;
        this._x2 = 0;
        this._y2 = 0;

        this._update();
    }

    Wire.prototype._update = function () {
        var length = Math.sqrt((this._x1 - this._x2) * (this._x1 - this._x2) + (this._y1 - this._y2) * (this._y1 - this._y2));
        var angle = Math.atan2(this._y2 - this._y1, this._x2 - this._x1) * 180.0 / Math.PI;

     //   length = Math.abs(length);

        var wireStyle = this._wire.style;
        wireStyle["width"] = Math.round(length) + 'px';
        wireStyle["left"] = Math.round(this._x1) + 'px';
        wireStyle["top"] = Math.round(this._y1) + 'px';
        wireStyle['-webkit-transform'] = 'rotate(' + angle + 'deg)';
        wireStyle['-moz-transform'] = 'rotate(' + angle + 'deg)';
        wireStyle['-ms-transform'] = 'rotate(' + angle + 'deg)';
        wireStyle['-o-transform'] = 'rotate(' + angle + 'deg)';
        wireStyle['transform'] = 'rotate(' + angle + 'deg)';
    };

    Wire.prototype.setStartAndEnd = function (x1, y1, x2, y2) {
        this._x1 = x1;
        this._y1 = y1;
        this._x2 = x2;
        this._y2 = y2;
        this._update();
    };

    Wire.prototype.setStart = function (x1, y1) {
        this._x1 = x1;
        this._y1 = y1;
        this._update();
    };

    Wire.prototype.setEnd = function (x2, y2) {
        this._x2 = x2;
        this._y2 = y2;
        this._update();
    };

    Wire.prototype.setColor = function (color) {
        this._wire.style.border = "solid 2px " + (color || "black");
    };

    Wire.prototype.setVisible = function (visible) {
        this._wire.style.visibility = visible ? "visible" : "hiidden";
    };


    function Dot(parentElement, cfg) {

        cfg = cfg || {};

        this._x = 0;
        this._y = 0;

        this._dot = document.createElement('div');
        var dot = this._dot;
        var style = dot.style;
        style["border-radius"] = 25 + "px";
        style.border = "solid 2px black";
        style.background = "lightgreen";
        style.position = "absolute";
        style["z-index"] = "2000001";
        style.width = 20 + "px";
        style.height = 20 + "px";
        style.visibility = "visible";
        style.top = 0 + "px";
        style.left = 0 + "px";
        style["pointer-events"] = "none";
        parentElement.appendChild(dot);

        this.setPos(cfg.x || 0, cfg.y || 0);
        this.setColor(cfg.fillColor);
        this.setBorderColor(cfg.borderColor);
    }

    Dot.prototype.setPos = function (x, y) {
        this._x = x;
        this._y = y;
        var style = this._dot.style;
        style["left"] = (Math.round(x) - 10) + 'px';
        style["top"] = (Math.round(y) - 10) + 'px';
    };

    Dot.prototype.setColor = function (color) {
        this._dot.style.background = color || "lightgreen";
    };

    Dot.prototype.setBorderColor = function (color) {
        this._dot.style.border = "solid 2px" + (color || "black");
    };

    Dot.prototype.setVisible = function (visible) {
        this._dot.style.visibility = visible ? "visible" : "hiidden";
    };

    xeogl.Ruler = xeogl.Component.extend({

        type: "xeogl.Ruler",

        _init: function (cfg) {

            var self = this;
            var scene = this.scene;

            this._originPin = new xeogl.Pin(this, cfg.origin);
            this._targetPin = new xeogl.Pin(this, cfg.target);

            this._originWorld = new Float32Array(3);
            this._targetWorld = new Float32Array(3);

            this._wp = new Float32Array(24); // World
            this._vp = new Float32Array(24); // View
            this._pp = new Float32Array(24); // Projection
            this._cp = new Int16Array(8); // Canvas

            var parentElement = document.body;

            this._originDot = new Dot(parentElement, {});
            this._targetDot = new Dot(parentElement, {});
            this._distWire = new Wire(parentElement, {color: "lightblue"});
            this._xAxisWire = new Wire(parentElement, {color: "red"});
            this._yAxisWire = new Wire(parentElement, {color: "green"});
            this._zAxisWire = new Wire(parentElement, {color: "blue"});

            this._wpDirty = false;
            this._vpDirty = false;
            this._cpDirty = false;

            this._originPin.on("worldPos", function (value) {
                self._originWorld.set(value || [0, 0, 0]);
                self._wpDirty = true;
                self._needUpdate();
            });

            this._targetPin.on("worldPos", function (value) {
                self._targetWorld.set(value || [0, 0, 0]);
                self._wpDirty = true;
                self._needUpdate();
            });

            this._onViewMatrix = scene.camera.on("viewMatrix", function () {
                self._vpDirty = true;
                self._needUpdate();
            });

            this._onProjectMatrix = scene.camera.on("projectMatrix", function () {
                self._cpDirty = true;
                self._needUpdate();
            });

            this.origin = cfg.origin;
            this.target = cfg.target;
            this.visible = cfg.visible;
        },

        _update: function () {

            //this._wp = new Float32Array(24); // World
            //this._vp = new Float32Array(24); // View
            //this._pp = new Float32Array(24); // Projection
            //this._cp = new Int16Array(8); // Canvas

            if (this._wpDirty) {

                this._wp[0] = this._originWorld[0];
                this._wp[1] = this._originWorld[1];
                this._wp[2] = this._originWorld[2];
                this._wp[3] = 1.0;

                this._wp[4] = this._targetWorld[0];
                this._wp[5] = this._originWorld[1];
                this._wp[6] = this._originWorld[2];
                this._wp[7] = 1.0;

                this._wp[8] = this._targetWorld[0];
                this._wp[9] = this._targetWorld[1];
                this._wp[10] = this._originWorld[2];
                this._wp[11] = 1.0;

                this._wp[12] = this._targetWorld[0];
                this._wp[13] = this._targetWorld[1];
                this._wp[14] = this._targetWorld[2];
                this._wp[15] = 1.0;

                this._wpDirty = false;
                this._vpDirty = true;
            }

            if (this._vpDirty) {

                xeogl.math.transformPositions4(this.scene.camera.view.matrix, this._wp, this._vp);

                this._vpDirty = false;
                this._cpDirty = true;
            }

            if (this._cpDirty) {

                xeogl.math.transformPositions4(this.scene.camera.project.matrix, this._vp, this._pp);

                var pp = this._pp;
                var cp = this._cp;

                var canvas = this.scene.canvas.canvas;
                var left = canvas.offsetLeft;
                var top = canvas.offsetTop;
                var aabb = this.scene.canvas.boundary;
                var canvasWidth = aabb[2];
                var canvasHeight = aabb[3];
                var j = 0;

                for (var i = 0, len = pp.length; i < len; i += 4) {
                    cp[j] = Math.floor((1 + pp[i + 0] / pp[i + 3]) * canvasWidth / 2);
                    cp[j + 1] = Math.floor((1 - pp[i + 1] / pp[i + 3]) * canvasHeight / 2);
                    j += 2;
                }

                this._originDot.setPos(cp[0], cp[1]);
                this._targetDot.setPos(cp[6], cp[7]);

                this._distWire.setStartAndEnd(cp[0], cp[1], cp[6], cp[7]);
                this._xAxisWire.setStartAndEnd(cp[0], cp[1], cp[2], cp[3]);
                this._yAxisWire.setStartAndEnd(cp[2], cp[3], cp[4], cp[5]);
                this._zAxisWire.setStartAndEnd(cp[4], cp[5], cp[6], cp[7]);

                this._cpDirty = false;
            }
        },

        _props: {

            /**
             * Origin Pin.
             * Fires an {{#crossLink "Ruler/origin:event"}}{{/crossLink}} event on change.
             * @property origin
             * @type {Pin}
             */
            origin: {

                get: function () {
                    return this._originPin;
                }
            },

            /**
             * Target Pin.
             * Fires an {{#crossLink "Ruler/target:event"}}{{/crossLink}} event on change.
             * @property target
             * @type {Pin}
             */
            target: {

                get: function () {
                    return this._targetPin;
                }
            },

            /**
             * Length of this Ruler.
             * Fires an {{#crossLink "Ruler/dist:event"}}{{/crossLink}} event on change.
             * @property dist
             * @default 0
             * @type {Number}
             */
            length: {

                get: function () {
                    return this._length;
                }
            },

            /**
             Indicates whether this Ruler is visible or not.

             Fires a {{#crossLink "Ruler/visible:event"}}{{/crossLink}} event on change.

             @property visible
             @default true
             @type Boolean
             */
            visible: {

                set: function (value) {

                    value = value !== false;

                    this._visible = value;

                    this._originDot.setVisible(this._visible);
                    this._targetDot.setVisible(this._visible);
                    this._distWire.setVisible(this._visible);
                    this._xAxisWire.setVisible(this._visible);
                    this._yAxisWire.setVisible(this._visible);
                    this._zAxisWire.setVisible(this._visible);

                    /**
                     Fired whenever this helper's {{#crossLink "Ruler/visible:property"}}{{/crossLink}} property changes.

                     @event visible
                     @param value {Boolean} The property's new value
                     */
                    this.fire("visible", this._visible);
                },

                get: function () {
                    return this._visible;
                }
            }
        },

        _getJSON: function () {
            return {
                origin: this._originPin.json,
                target: this._targetPin.json,
                visible: this._visible
            };
        },

        _destroy: function () {
            if (this._onViewMatrix) {
                this.scene.camera.off(this._onViewMatrix);
            }
            if (this._onProjectMatrix) {
                this.scene.camera.off(this._onProjectMatrix);
            }
            if (this._onTick) {
                this.scene.camera.off(this._onTick);
            }
        }
    });
})();
